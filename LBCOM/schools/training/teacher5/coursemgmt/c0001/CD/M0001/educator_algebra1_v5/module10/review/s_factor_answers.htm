Solutions - Solving Quadratic Equations by Factoring

  1. x2 - 7x + 12 = 0
    (x - 4)(x - 3) = 0
    x - 4 = 0 or x - 3 = 0
    x = 4 or x = 3

  2. 3 a2 = 9a
    3a2 - 9a = 0
    3a(a - 3) = 0
    3a = 0 or a - 3 =0
    a = or a = 3
    a = 0, a = 3

  3. b2 + 6b = -5
    b2 + 6b + 5 = 0
    (b + 5)(b + 1) = 0
    b + 5 = 0 or b + 1 = 0
    b = -5, b = -1
